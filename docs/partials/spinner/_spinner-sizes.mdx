import { Root as NewfoldRoot, Spinner } from "@newfold/ui-component-library";

#### 1. Size 3 (Small)
To render size 3 spinner pass `'3'` to the `size` prop.

<NewfoldRoot>
    <div className="nfd-flex nfd-items-center nfd-gap-3">
        <Spinner size="3" />
        <Spinner 
            size="3"
            variant="primary"
        />
        <div className="nfd-inline-block nfd-bg-[#2B2B2B] nfd-p-1.5 nfd-rounded">
            <Spinner
                size="3"
                variant="white" 
            />
        </div>
    </div>
</NewfoldRoot>

```jsx
<div className="nfd-flex nfd-items-center nfd-gap-3">
    <Spinner size="3" />
    <Spinner 
        size="3"
        variant="primary"
    />
    <div className="nfd-inline-block nfd-bg-[#2B2B2B] nfd-p-1.5 nfd-rounded">
        <Spinner
            size="3"
            variant="white" 
        />
    </div>
</div>
```
