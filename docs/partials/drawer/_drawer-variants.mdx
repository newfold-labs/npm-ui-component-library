import { Root as NewfoldRoot } from "@newfold/ui-component-library";
import { DrawerDemo } from ".";

#### Classic
The `classic` variant is the default variant.

<NewfoldRoot>
    <DrawerDemo variant="classic" buttonText="Classic Drawer" />
</NewfoldRoot>

```jsx
<Drawer variant="classic">
    <Drawer.Panel>
        ...
    </Drawer.Panel>
</Drawer>
```

#### Offset
The `offset` variant renders the drawer with spacing around the panel.

<NewfoldRoot>
    <DrawerDemo variant="offset" buttonText="Offset Drawer" />
</NewfoldRoot>

```jsx
<Drawer variant="offset">
    <Drawer.Panel>
        ...
    </Drawer.Panel>
</Drawer>
```
