import { Root as NewfoldRoot } from "@newfold/ui-component-library";
import { DrawerDemo } from ".";

#### Right
The `right` position is the default position. It will render the drawer on the right side of the page.

<NewfoldRoot>
    <DrawerDemo position="right" buttonText="Right Drawer" />
</NewfoldRoot>

```jsx
<Drawer position="right">
    <Drawer.Panel>
        ...
    </Drawer.Panel>
</Drawer>
```

#### Left
The `left` position will render the drawer on the left side of the page.

<NewfoldRoot>
    <DrawerDemo position="left" buttonText="Left Drawer" />
</NewfoldRoot>

```jsx
<Drawer position="left">
    <Drawer.Panel>
        ...
    </Drawer.Panel>
</Drawer>
```
