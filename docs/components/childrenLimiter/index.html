Usage​

import { ChildrenLimiter } from "@newfold/ui-component-library";

<ChildrenLimiter 
    limit={5}
    renderButton={ ( { show, toggle, ariaProps } ) => {
        return <Button className="nfd-my-1.5" onClick={ toggle } { ...ariaProps }>{ show ? "Show Less" : "Show More" }</Button>; } 
    }>
    <p>0</p>
    <p>1</p>
    <p>2</p>
    <p>3</p>
    <p>4</p>
    <p>5</p>
    <p>6</p>
    <p>7</p>
    <p>8</p>
    <p>9</p>
</ChildrenLimiter>
